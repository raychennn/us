# 使用輕量級 Python 映像
FROM python:3.9-slim

# 設定工作目錄
WORKDIR /app

# --- 關鍵修正 ---
# 安裝系統依賴 (GCC, Make 等編譯工具)
# 這是為了解決 pandas 或 numpy 需要從源碼編譯時發生的 "ninja" 或 "metadata" 錯誤
RUN apt-get update && \
    apt-get install -y build-essential python3-dev curl && \
    rm -rf /var/lib/apt/lists/*

# 複製 requirements.txt 並安裝
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# 複製所有程式碼到容器
COPY . .

# 設定時區 (確保 Log 時間正確)
ENV TZ=Asia/Taipei

# 執行程式
CMD ["python", "main.py"]
